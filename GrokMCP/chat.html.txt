        <div class="messages" id="messages"></div>
        <div class="input-area">
            <input type="text" id="prompt" placeholder="Ask anything MSP-related... (e.g. pricing for Winter Park dental office)">
            <button onclick="sendMessage()">Send</button>
        </div>
    </div>

    <script>
        async function sendMessage() {
            const input = document.getElementById('prompt');
            const promptText = input.value.trim();
            if (!promptText) return;

            addMessage(promptText, 'user');
            input.value = '';

            try {
                const res = await fetch('http://localhost:8000/mcp/query', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ prompt: promptText })
                });
                const data = await res.json();
                addMessage(data.response || "No response received", 'assistant');
            } catch (e) {
                addMessage("Error connecting to local Grok MCP. Make sure the server is running.", 'assistant');
            }
        }

        function addMessage(text, type) {
            const div = document.createElement('div');
            div.className = `message ${type}`;
            div.textContent = text;
            document.getElementById('messages').appendChild(div);
            div.scrollIntoView({ behavior: "smooth" });
        }

        document.getElementById('prompt').addEventListener('keypress', e => {
            if (e.key === 'Enter') sendMessage();
        });
    </script>
